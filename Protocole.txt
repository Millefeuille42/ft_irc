Client :
	Pseudonyme -> maximum 9 caractères
	Vrai nom + Nom d'utilisateur + Serveur Connecté

Channel :
	Commence par & ou # -> Maximum 200 caractères (Interdit -> ' ', '^G', ',')
	# -> Connu de tous les serveurs connectés au réseau.
	& -> Accessible uniquement aux clients du serveur où le canal existe.
	Modes :
		Invités uniquement (+i)
	Utilisateur créant le channel = Opérateur du channel.

Opérateurs de channel :
   	Commandes :
   		KICK -> Éjecte un client du canal
   		MODE -> Change le mode du canal
   		INVITE -> Invite un client dans le canal (mode +i)
   		TOPIC -> (Change le titre du canal (mode +t))
   	Identification :
   		@ devant le pseudonyme en réponse aux commandes ->
   			NAMES
   			WHO
   			WHOIS


Commandes :
	Connexion :
		PASS <mot de passe>
			Définit le mot de passe avant toute tentative de connexion.
		NICK <pseudonyme> [<compteur de distance>]
			Définit le pseudonyme d'un utilisateur.
		USER <nom d'utilisateur> <hôte> <nom de serveur> <nom réel>

		OPER <utilisateur> <mot de passe>

		QUIT [<Message de départ>]

	Canaux :
		JOIN <canal>{,<canal>} [<clé>{,<clé>}]
		PART <canal>{,< canal >}
		MODE <canal> {[+|-]|o|p|s|i|t|n|b|v} [<limite>] [<utilisateur>] [<masque de bannissement >]
		MODE <pseudonyme> {[+|-]|i|w|s|o}
		TOPIC <canal> [<sujet>]
		NAMES [<canal>{,<canal>}]
		LIST [<canal>{,<canal>}
		INVITE <pseudonyme> <canal>
		KICK <canal> <utilisateur> [<commentaire>]

	Serveurs :
		VERSION
		TIME
		INFO

	Messages :
		PRIVMSG <destinataire>{,<destinataire>} <texte à envoyer >

	Requêtes :
		WHO [<nom> [<o>]]
		//WHOIS [<serveur>] <masque de pseudo>[,<masque de pseudo>[,...]]

	Divers :
		 KILL <pseudonyme> <commentaire>
		 PING <serveur1> [<serveur2>]
		 PONG <démon> [<démon2>]
		 ERROR < message d'erreur>

Optionnels :
	//AWAY [message]
	//USERS [<serveur>]
	//WALLOPS [message]
